堆排序（数组模拟）

2022/9/4 算法 ALGO C/C++ others

堆是特殊的队列，特殊的点就在于它可以依照元素的优先级取出，而不是元素入队的先后顺序。

堆常表现为完全二叉树的形式，因为高度为h的完全二叉树有结点 2(k-1)到2k-1个，且结点排布极共规律，因此，通常不必用指针，而是用数组来实现堆的存储。

堆排序核心操作，建堆

void down(int u){
int t=u;

//构建平衡二叉搜索树
if( 2*u<=cnt && h[2*u]<h[t] ) t=2*u;
if( 2*u+1 <= cnt && h[2*u+1]<h[t]) t=2*u+1;
//代表树的位置已经发生变化,递归直到u==t时结束
if(u!=t){
    swap(h[u],h[t]);
    down(t);
}
}

只要理解平衡二叉树的原理就好

完整代码

#include <iostream>
#include <cstring>
#include <algorithm>

using namespace std;

const int N = 1e5+10;

int n,m;
//cnt记录结点数，h数组模拟树
int cnt,h[N];

void down(int u){
int t=u;

if( 2*u<=cnt && h[2*u]<h[t] ) t=2*u;
if( 2*u+1 <= cnt && h[2*u+1]<h[t]) t=2*u+1;
if(u!=t){
    swap(h[u],h[t]);
    down(t);
}
}

int main()
{
    scanf("%d%d", &n ,&m );
    for (int i = 1; i <= n; i ++ ) 
	scanf("%d", &h[i]);

	//更新cnt
	cnt=n;

	//从树顶开始向下递归
	for (int i = n/2; i ; i -- )
	down(i);

while (m -- ){
    printf("%d ",h[1]);
    //保证每次输出的顶都是最小的
    h[1]=h[cnt--];
    down(1);
}
}
